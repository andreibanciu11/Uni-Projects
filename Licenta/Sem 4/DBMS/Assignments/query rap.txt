CREATE TABLE Labels (
    LabelID INT PRIMARY KEY,
    LabelName VARCHAR(255),
    Country VARCHAR(255),
    City VARCHAR(255),
    DateOfEstablishment DATE,
    Genre VARCHAR(255)
);

-- Artists table
CREATE TABLE Artists (
    ArtistID INT PRIMARY KEY,
    ArtistName VARCHAR(255),
    Country VARCHAR(255),
    DOB DATE,
    NumberOfStreams INT,
    SignedLabelID INT,
    FOREIGN KEY (SignedLabelID) REFERENCES Labels(LabelID)
);

-- Concerts table
CREATE TABLE Concerts (
    ConcertID INT PRIMARY KEY,
    ConcertName VARCHAR(255),
    ConcertDate DATE,
    Country VARCHAR(255),
    City VARCHAR(255),
    SignedArtistID INT,
    FOREIGN KEY (SignedArtistID) REFERENCES Artists(ArtistID)
);

-- Producers table
CREATE TABLE Producers (
    ProducerID INT PRIMARY KEY,
    ProducerName VARCHAR(255),
    Country VARCHAR(255),
    DOB DATE
);

-- Songs table
CREATE TABLE Songs (
    SongID INT PRIMARY KEY,
    Title VARCHAR(255),
    DateOfRelease DATE,
    SignedProducerID INT,
    FOREIGN KEY (SignedProducerID) REFERENCES Producers(ProducerID)
);

-- Performing table
CREATE TABLE Performing (
    ArtistID INT,
    SongID INT,
    Featuring INT,
    PRIMARY KEY (ArtistID, SongID),
    FOREIGN KEY (ArtistID) REFERENCES Artists(ArtistID),
    FOREIGN KEY (SongID) REFERENCES Songs(SongID)
);

-- Platforms table
CREATE TABLE Platforms (
    PlatformID INT PRIMARY KEY,
    PlatformName VARCHAR(255),
    NumberOfUsers INT
);

-- PlayingPlatform table
CREATE TABLE PlayingPlatform (
    PlatformID INT,
    SongID INT,
    PRIMARY KEY (PlatformID, SongID),
    FOREIGN KEY (PlatformID) REFERENCES Platforms(PlatformID),
    FOREIGN KEY (SongID) REFERENCES Songs(SongID)
);

-- Radios table
CREATE TABLE Radio (
    RadioFMID INT PRIMARY KEY,
    RadioName VARCHAR(255),
    NumberOfUsers INT
);

-- PlayingRadio table
CREATE TABLE PlayingRadio (
    RadioID INT,
    SongID INT,
    PRIMARY KEY (RadioID, SongID),
    FOREIGN KEY (RadioID) REFERENCES Radio(RadioFMID),
    FOREIGN KEY (SongID) REFERENCES Songs(SongID)
);
